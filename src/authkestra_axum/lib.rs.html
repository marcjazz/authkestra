52/// The extractor for a validated session.
53pub struct AuthSession(pub Session);
54
55impl<S> FromRequestParts<S> for AuthSession
56where
57    S: Send + Sync,
58    Arc<dyn SessionStore>: FromRef<S>,
59    SessionConfig: FromRef<S>,
60{
61    type Rejection = AuthkestraAxumError;
62
63    async fn from_request_parts(parts: &mut Parts, state: &S) -> Result<Self, Self::Rejection> {
64        let session_store = Arc::<dyn SessionStore>::from_ref(state);
65        let session_config = SessionConfig::from_ref(state);
66        let cookies = Cookies::from_request_parts(parts, state)
67            .await
68            .map_err(|e| AuthkestraAxumError::Internal(e.1.to_string()))?;
69
70        let session = helpers::get_session(&session_store, &session_config, &cookies).await?;
71
72        Ok(AuthSession(session))
73    }
74}

76/// The extractor for a validated JWT.
77///
78/// Expects an `Authorization: Bearer <token>` header.
79pub struct AuthToken(pub authkestra_token::Claims);
80
81impl<S> FromRequestParts<S> for AuthToken
82where
83    S: Send + Sync,
84    Arc<TokenManager>: FromRef<S>,
85{
86    type Rejection = AuthkestraAxumError;
87
88    async fn from_request_parts(parts: &mut Parts, state: &S) -> Result<Self, Self::Rejection> {
89        let token_manager = Arc::<TokenManager>::from_ref(state);
90        let token = helpers::get_token(parts, &token_manager).await?;
91        Ok(AuthToken(token))
92    }
93}
