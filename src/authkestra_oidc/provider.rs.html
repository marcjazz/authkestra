39impl OidcProvider {
40    /// Creates a new provider by performing discovery
41    pub async fn discover(
42        client_id: String,
43        client_secret: String,
44        redirect_uri: String,
45        issuer_url: &str,
46    ) -> Result<Self, OidcError> {
47        let client = reqwest::Client::new();
48        let metadata = ProviderMetadata::discover(issuer_url, &client).await?;
49        Ok(Self {
50            client_id,
51            client_secret,
52            redirect_uri,
53            metadata,
54            http_client: client,
55        })
56    }
57
58    pub fn metadata(&self) -> &ProviderMetadata {
59        &self.metadata
60    }

63#[async_trait]
64impl OAuthProvider for OidcProvider {
65    fn provider_id(&self) -> &str {
66        "oidc"
67    }
68
69    fn get_authorization_url(
70        &self,
71        state: &str,
72        scopes: &[&str],
73        code_challenge: Option<&str>,
74    ) -> String {
75        let mut full_scopes = scopes.to_vec();
76        if !full_scopes.contains(&"openid") {
77            full_scopes.push("openid");
78        }
79
80        let scope_param = full_scopes.join(" ");
81
82        let mut url = format!(
83            "{}?client_id={}&redirect_uri={}&response_type=code&state={}&scope={}",
84            self.metadata.authorization_endpoint,
85            self.client_id,
86            urlencoding::encode(&self.redirect_uri),
87            state,
88            urlencoding::encode(&scope_param)
89        );
90
91        if let Some(challenge) = code_challenge {
92            url.push_str(&format!(
93                "&code_challenge={}&code_challenge_method=S256",
94                challenge
95            ));
96        }
97
98        url
99    }
100
101    async fn exchange_code_for_identity(
102        &self,
103        code: &str,
104        code_verifier: Option<&str>,
105    ) -> Result<(Identity, OAuthToken), AuthError> {
106        // 1. Exchange code for tokens
107        let mut params = vec![
108            ("grant_type", "authorization_code".to_string()),
109            ("code", code.to_string()),
110            ("redirect_uri", self.redirect_uri.clone()),
111            ("client_id", self.client_id.clone()),
112            ("client_secret", self.client_secret.clone()),
113        ];
114
115        if let Some(verifier) = code_verifier {
116            params.push(("code_verifier", verifier.to_string()));
117        }
118
119        let token_response = self
120            .http_client
121            .post(&self.metadata.token_endpoint)
122            .form(&params)
123            .send()
124            .await
125            .map_err(|_| AuthError::Network)?
126            .json::<OidcTokenResponse>()
127            .await
128            .map_err(|e| AuthError::Provider(format!("Failed to parse token response: {}", e)))?;
129
130        let id_token = token_response
131            .id_token
132            .ok_or_else(|| AuthError::Token("Missing id_token in response".to_string()))?;
133
134        // 2. Fetch JWKS
135        let jwks = Jwks::fetch(&self.metadata.jwks_uri, &self.http_client)
136            .await
137            .map_err(AuthError::from)?;
138
139        // 3. Decode and Validate ID Token
140        let header = decode_header(&id_token)
141            .map_err(|e| AuthError::Token(format!("Invalid ID Token header: {}", e)))?;
142
143        let jwk = jwks
144            .find_key(header.kid.as_deref())
145            .ok_or_else(|| AuthError::Token("No matching key found in JWKS".to_string()))?;
146
147        let decoding_key = jwk.to_decoding_key().map_err(AuthError::from)?;
148
149        let mut validation = Validation::new(Algorithm::RS256);
150        validation.set_issuer(std::slice::from_ref(&self.metadata.issuer));
151        validation.set_audience(std::slice::from_ref(&self.client_id));
152
153        let token_data = decode::<Claims>(&id_token, &decoding_key, &validation)
154            .map_err(|e| AuthError::Token(format!("ID Token validation failed: {}", e)))?;
155
156        let claims = token_data.claims;
157
158        // 4. Construct Identity
159        let mut attributes = HashMap::new();
160        if let Some(picture) = claims.picture {
161            attributes.insert("picture".to_string(), picture);
162        }
163
164        let identity = Identity {
165            provider_id: "oidc".to_string(), // Could be parameterized or inferred from issuer
166            external_id: claims.sub,
167            email: claims.email,
168            username: claims.name,
169            attributes,
170        };
171
172        let token = OAuthToken {
173            access_token: token_response.access_token,
174            token_type: token_response.token_type,
175            expires_in: token_response.expires_in,
176            refresh_token: token_response.refresh_token,
177            scope: token_response.scope,
178            id_token: Some(id_token),
179        };
180
181        Ok((identity, token))
182    }
