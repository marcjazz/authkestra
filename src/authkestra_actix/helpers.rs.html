12pub fn to_actix_same_site(ss: authkestra_core::SameSite) -> actix_web::cookie::SameSite {
13    match ss {
14        authkestra_core::SameSite::Lax => actix_web::cookie::SameSite::Lax,
15        authkestra_core::SameSite::Strict => actix_web::cookie::SameSite::Strict,
16        authkestra_core::SameSite::None => actix_web::cookie::SameSite::None,
17    }
18}

20pub fn create_actix_cookie<'a>(config: &SessionConfig, value: String) -> Cookie<'a> {
21    let mut builder = Cookie::build(config.cookie_name.clone(), value)
22        .path(config.path.clone())
23        .secure(config.secure)
24        .http_only(config.http_only)
25        .same_site(to_actix_same_site(config.same_site));
26
27    if let Some(max_age) = config.max_age {
28        builder = builder.max_age(actix_web::cookie::time::Duration::seconds(
29            max_age.num_seconds(),
30        ));
31    }
32    builder.finish()
33}

35/// Helper to initiate the OAuth2 login flow.
36///
37/// This generates the authorization URL and sets a CSRF state cookie.
38pub fn initiate_oauth_login<P, M>(
39    flow: &OAuth2Flow<P, M>,
40    session_config: &SessionConfig,
41    scopes: &[&str],
42) -> HttpResponse
43where
44    P: OAuthProvider,
45    M: authkestra_core::UserMapper,
46{
47    initiate_oauth_login_erased(flow, session_config, scopes)
48}

50pub fn initiate_oauth_login_erased(
51    flow: &dyn ErasedOAuthFlow,
52    session_config: &SessionConfig,
53    scopes: &[&str],
54) -> HttpResponse {
55    let pkce = Pkce::new();
56    let (url, csrf_state) = flow.initiate_login(scopes, Some(&pkce.code_challenge));
57
58    let cookie_name = format!("authkestra_flow_{}", csrf_state);
59
60    let cookie = Cookie::build(cookie_name, pkce.code_verifier)
61        .path("/")
62        .http_only(true)
63        .same_site(actix_web::cookie::SameSite::Lax)
64        .secure(session_config.secure)
65        .max_age(actix_web::cookie::time::Duration::minutes(15))
66        .finish();
67
68    HttpResponse::Found()
69        .insert_header((header::LOCATION, url))
70        .cookie(cookie)
71        .finish()
72}

74/// Helper to handle the OAuth2 callback and create a server-side session.
75pub async fn handle_oauth_callback<P, M>(
76    req: HttpRequest,
77    flow: &OAuth2Flow<P, M>,
78    params: OAuthCallbackParams,
79    store: Arc<dyn SessionStore>,
80    config: SessionConfig,
81    success_url: &str,
82) -> Result<HttpResponse, actix_web::Error>
83where
84    P: OAuthProvider + Send + Sync,
85    M: authkestra_core::UserMapper + Send + Sync,
86{
87    handle_oauth_callback_erased(req, flow, params, store, config, success_url).await

90pub async fn handle_oauth_callback_erased(
91    req: HttpRequest,
92    flow: &dyn ErasedOAuthFlow,
93    params: OAuthCallbackParams,
94    store: Arc<dyn SessionStore>,
95    config: SessionConfig,
96    success_url: &str,
97) -> Result<HttpResponse, actix_web::Error> {
98    let cookie_name = format!("authkestra_flow_{}", params.state);
99    let pkce_verifier = req
100        .cookie(&cookie_name)
101        .map(|c| c.value().to_string())
102        .ok_or_else(|| {
103            actix_web::error::ErrorUnauthorized("CSRF validation failed or session expired")
104        })?;
105
106    // Exchange code
107    let (mut identity, token) = flow
108        .finalize_login(
109            &params.code,
110            &params.state,
111            &params.state, // We use the state itself as expected_state
112            Some(&pkce_verifier),
113        )
114        .await
115        .map_err(|e| {
116            actix_web::error::ErrorUnauthorized(format!("Authentication failed: {}", e))
117        })?;
118
119    // Store tokens in identity attributes for convenience
120    identity
121        .attributes
122        .insert("access_token".to_string(), token.access_token);
123    if let Some(expires_in) = token.expires_in {
124        let expires_at = chrono::Utc::now().timestamp() + expires_in as i64;
125        identity
126            .attributes
127            .insert("expires_at".to_string(), expires_at.to_string());
128    }
129    if let Some(rt) = token.refresh_token {
130        identity.attributes.insert("refresh_token".to_string(), rt);
131    }
132    println!("Created identity: {:?}", identity);
133
134    let session_duration = config.max_age.unwrap_or(chrono::Duration::hours(24));
135    let session = Session {
136        id: uuid::Uuid::new_v4().to_string(),
137        identity,
138        expires_at: chrono::Utc::now() + session_duration,
139    };
140
141    store.save_session(&session).await.map_err(|e| {
142        actix_web::error::ErrorInternalServerError(format!("Failed to save session: {}", e))
143    })?;
144
145    let cookie = create_actix_cookie(&config, session.id);
146
147    // Remove the flow cookie
148    let remove_cookie = Cookie::build(cookie_name, "")
149        .path("/")
150        .secure(config.secure)
151        .max_age(actix_web::cookie::time::Duration::ZERO)
152        .finish();
153
154    Ok(HttpResponse::Found()
155        .insert_header((header::LOCATION, success_url))
156        .cookie(cookie)
157        .cookie(remove_cookie)
158        .finish())
159}

161pub async fn actix_login_handler(
162    path: web::Path<String>,
163    authkestra: web::Data<Authkestra>,
164) -> impl actix_web::Responder {
165    let provider = path.into_inner();
166    let flow = match authkestra.providers.get(&provider) {
167        Some(f) => f,
168        None => return HttpResponse::NotFound().body(format!("Provider {} not found", provider)),
169    };
170
171    initiate_oauth_login_erased(flow.as_ref(), &authkestra.session_config, &[])
172}

174pub async fn actix_callback_handler(
175    req: HttpRequest,
176    path: web::Path<String>,
177    authkestra: web::Data<Authkestra>,
178    params: web::Query<OAuthCallbackParams>,
179) -> actix_web::Result<impl actix_web::Responder> {
180    let provider = path.into_inner();
181    let flow = match authkestra.providers.get(&provider) {
182        Some(f) => f,
183        None => {
184            return Ok(HttpResponse::NotFound().body(format!("Provider {} not found", provider)))
185        }
186    };
187
188    handle_oauth_callback_erased(
189        req,
190        flow.as_ref(),
191        params.into_inner(),
192        authkestra.session_store.clone(),
193        authkestra.session_config.clone(),
194        "/",
195    )
196    .await
197}

212/// Helper to handle logout by deleting the session from the store and clearing the cookie.
213pub async fn logout(
214    req: HttpRequest,
215    store: Arc<dyn SessionStore>,
216    config: SessionConfig,
217    redirect_to: &str,
218) -> Result<HttpResponse, actix_web::Error> {
219    let session_id = req
220        .cookie(&config.cookie_name)
221        .map(|c| c.value().to_string());
222
223    if let Some(id) = session_id {
224        store
225            .delete_session(&id)
226            .await
227            .map_err(|e| actix_web::error::ErrorInternalServerError(e.to_string()))?;
228    }
229
230    let remove_cookie = create_actix_cookie(&config, "".to_string());
231
232    Ok(HttpResponse::Found()
233        .insert_header((header::LOCATION, redirect_to))
234        .cookie(remove_cookie)
235        .finish())
